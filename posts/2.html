2 判断语句和循环语句

2.1 判断语句

if true:
   print("True") #缩进统一
else:
   print("False")

例1.成绩评级

score = int(input("输入成绩："))

if score >=90 and score <=100:
	print("本次考试，等级为A")
elif	score >=80 and score <90:
	print("本次考试，等级为B")
elif	score >=70 and score <80:
	print("本次考试，等级为C")
elif	score >=60 and score <70:
	print("本次考试，等级为D")
elif	score >=0 and score <60:
#else:
	print("本次考试，成绩为F")

例3.石头剪子布

import random #引入随机库
y = int(input("请输入：剪刀（0），石头（1），布（2）："))
x = random.randint(0,2)
if y == 1 :
    print("您输入的为：石头（1）")
    if x == 1:
        print("随机生成的数字为：1")
        print("平局")
    elif x == 0:
        print("随机生成的数字为：0")
        print("您赢了")
    elif x == 2:
        print("随机生成的数字为：2")
        print("您输了")
elif y == 0:
    print("您输入的为：剪刀（0）")
    if x == 1:
        print("随机生成的数字为：1")
        print("您输了")
    elif x == 0:
        print("随机生成的数字为：0")
        print("平局")
    elif x == 2:
        print("随机生成的数字为：2")
        print("您赢了")
elif y == 2:
    print("您输入的为：布（2）")
    if x == 1:
        print("随机生成的数字为：1")
        print("您赢了")
    elif x == 0:
        print("随机生成的数字为：0")
        print("您输了")
    elif x == 2:
        print("随机生成的数字为：2")
        print("平局")

2.2 循环语句

2.2.1 for循环

for i in range(5):
	print(i)
for i in range(0,10,3):
	print(i)
#从0开始到10结束，步进值为3，每次加3
for i in range(-10,-100,-30):
	print(i)
'''
-10
-40
-70
'''
name="chengdu"
for x in name:
    print(x,end="\t")
#c h e n g d u
a=["aa","bb","cc","dd"]#列表
for i in range(len(a)):#遍历字符串 a的长度
	print(i,a[i])

2.2.2 while 循环

i = 0
while i < 5:
	print("当前是第%d次循环："%(i+1))
	print("i=%d"%i)
	i += 1

例4. 1到100的和

n=100
sum=0
i=1
while i <= n:
    sum=sum+i
    i += 1
print("1到%d的和为：%d"%(n,sum))
sum=0
for i in range(1,101):
    sum=sum+i
print("1到%d的值为：%d"%(i,sum))
count = 0
while count < 5:
    print(count,"小于5")
    count += 1
else:
    print(count,"大于或等于5")
